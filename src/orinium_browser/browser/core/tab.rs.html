25/// Tab はブラウザで開かれた 1 つのページを表す構造体です。
26///
27/// 主な責務:
28/// - 現在表示しているページのタイトルの保持
29/// - ページ内容を扱う WebView の保持
30///
31/// WebView が「ページそのもの」の状態を管理するのに対し、
32/// Tab は UI 上のタブとしてのメタ情報（タイトルなど）を管理します。
33///
34/// TODO:
35/// - ページの状態（Error、loading）の管理を追加
36pub struct Tab {
37    title: Option<String>,
38    base_url: Option<Url>,
39    docment_url: Option<Url>,
40    webview: Option<WebView>,
41    state: TabState,
42}
43
44impl Default for Tab {
45    fn default() -> Self {
46        Self::new()
47    }
48}

50impl Tab {
51    pub fn new() -> Self {
52        Self {
53            title: None,
54            base_url: None,
55            docment_url: None,
56            webview: None,
57            state: TabState::Loading,
58        }
59    }

61    /// Tab 内の状態を 1 ステップ進める
62    ///
63    /// - WebView.tick() を呼び出す
64    /// - 発生した Task を BrowserApp に返す
65    pub fn tick(&mut self) -> Vec<TabTask> {
66        let mut tasks = Vec::new();
67        let Some(wv) = self.webview.as_mut() else {
68            return tasks;
69        };
70
71        for task in wv.tick() {
72            match task {
73                WebViewTask::Fetch { url, kind } => {
74                    log::info!("Fetch requested in Tab: url={}", url);
75                    tasks.push(TabTask::Fetch { url, kind });
76                }
77                WebViewTask::AskTabHtml => {
78                    tasks.push(TabTask::Fetch {
79                        url: self.docment_url.as_ref().unwrap().clone(),
80                        kind: FetchKind::Html,
81                    });
82                }
83            }
84        }
85
86        if wv.needs_redraw() {
87            tasks.push(TabTask::NeedsRedraw);
88        }
89
90        tasks
91    }

93    /// BrowserApp から CSS fetch 完了を通知
94    pub fn on_css_fetched(&mut self, css: String) {
95        log::info!("CSS fetched in Tab");
96        if let Some(webview) = self.webview.as_mut() {
97            webview.on_css_fetched(css);
98        }
99    }
100
101    /// BrowserApp からの HTML fetch 完了を通知
102    pub fn on_fetch_succeeded_html(&mut self, html: String) {
103        let Some(wv) = self.webview.as_mut() else {
104            return;
105        };
106
107        wv.on_html_fetched(html, self.docment_url.as_ref().unwrap().clone());
108        self.title = wv.title().cloned();
109        let base_url = wv.base_url().unwrap().clone();
110        log::info!("HTML fetched, base_url={}", base_url);
111        self.base_url = Some(base_url);
112
113        if let TabState::Error(TabError::NetworkError(err), url_opt) = &self.state {
114            let error_message = match url_opt {
115                Some(url) => format!("Failed to load {}: {}", url, err),
116                None => format!("Failed to load page: {}", err),
117            };
118
119            let error_message_element = wv
120                .document_info()
121                .unwrap()
122                .dom
123                .get_elements_by_class_name("error-message");
124            let error_message_element = error_message_element.first().unwrap();
125            let new_child = TreeNode::new(HtmlNodeType::Text(error_message));
126            TreeNode::replace_child(error_message_element, 0, new_child);
127
128            // Update page to show error message
129            // This is a stub implementation for now as you can see in WebView.update_page().
130            wv.update_page();
131        } else {
132            self.state = TabState::Loaded;
133        }
134    }
135
136    pub fn on_fetch_succeeded_css(&mut self, css: String) {
137        let Some(wv) = self.webview.as_mut() else {
138            return;
139        };
140
141        wv.on_css_fetched(css);
142    }

144    /// Display error page on fetch failure
145    pub fn on_fetch_failed(&mut self, err: BrowserNetworkError, failed_url: Url) {
146        self.navigate("resource:///error.html".parse().unwrap());
147        self.state = TabState::Error(TabError::NetworkError(err), Some(failed_url));
148    }
149
150    pub fn navigate(&mut self, url: Url) {
151        self.docment_url = Some(url.clone());
152        let mut webview = WebView::new();
153        webview.navigate();
154        self.webview = Some(webview);
155        self.state = TabState::Loading;
156    }
157
158    pub fn move_to(&mut self, href: &str) {
159        let base_url = match self.base_url.as_ref() {
160            Some(u) => u,
161            None => return,
162        };
163
164        let url = super::webview::resolve_url(base_url, href).unwrap();
165
166        // navigate と同じ扱い
167        self.navigate(url)
168    }
169
170    pub fn relayout(&mut self, viewport: (f32, f32)) {
171        if let Some(wv) = self.webview.as_mut() {
172            wv.relayout(viewport);
173        }
174    }

176    /// Returns layout_and_info

184    /// Returns title of the document

193
194    pub fn layout_and_info(&self) -> Option<(&LayoutNode, &InfoNode)> {
195        self.webview.as_ref().and_then(|wv| wv.layout_and_info())
196    }
197
198    pub fn needs_redraw(&self) -> bool {
199        self.webview
200            .as_ref()
201            .map(|wv| wv.needs_redraw())
202            .unwrap_or(false)
