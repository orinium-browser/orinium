23/// glyphon使ったテキストレンダラー
24pub struct TextRenderer {
25    /// glyphonのテキストブラシ
26    brush: TextBrush,
27    /// ビューポート情報
28    viewport: Viewport,
29    /// glyphonのテキストアトラス
30    atlas: TextAtlas,
31    /// rasterize 結果のキャッシュ
32    swash_cache: SwashCache,
33    font_sys: FontSystem,
34}
35
36impl TextRenderer {
37    /// 情報を渡してシステムフォントから初期化する
38    pub fn new_from_device(
39        device: &wgpu::Device,
40        queue: &wgpu::Queue,
41        format: wgpu::TextureFormat,
42    ) -> anyhow::Result<Self> {
43        // 後々環境変数とかに設定しているときに使えるようにしてます
44        if let Ok(p) = env::var("ORINIUM_FONT")
45            && let Ok(bytes) = std::fs::read(&p)
46        {
47            return Self::new_from_bytes(device, queue, format, bytes);
48        }
49
50        for p in font::system_font_candidates()? {
51            if let Ok(bytes) = std::fs::read(p) {
52                // build brush from bytes
53                return Self::new_from_bytes(device, queue, format, bytes);
54            }
55        }
56
57        anyhow::bail!("no system font found");
58    }
59
60    pub fn new_with_fontsys(
61        device: &wgpu::Device,
62        queue: &wgpu::Queue,
63        format: wgpu::TextureFormat,
64        font_sys: FontSystem,
65    ) -> anyhow::Result<Self> {
66        let cache = Cache::new(device);
67        let mut atlas = TextAtlas::new(device, queue, &cache, format);
68        let multisample = wgpu::MultisampleState {
69            count: 1,                         // MSAA 無効
70            mask: !0,                         // 全サンプル有効
71            alpha_to_coverage_enabled: false, // glyphon は距離場なので不要
72        };
73        let brush = TextBrush::new(&mut atlas, device, multisample, None);
74
75        let viewport = Viewport::new(device, &cache);
76
77        let swash_cache = SwashCache::new();
78
79        Ok(Self {
80            brush,
81            atlas,
82            font_sys,
83            viewport,
84            swash_cache,
85        })
86    }
87
88    /// フォントバイト列から生成するコンストラクタ
89    pub fn new_from_bytes(
90        device: &wgpu::Device,
91        queue: &wgpu::Queue,
92        format: wgpu::TextureFormat,
93        font_bytes: Vec<u8>,
94    ) -> anyhow::Result<Self> {
95        let font_source = Arc::new(font_bytes);
96        let font = fontdb::Source::Binary(font_source);
97        let font_sys = FontSystem::new_with_fonts(vec![font]);
98        Self::new_with_fontsys(device, queue, format, font_sys)
99    }

101    /// Create a cosmic-text `Buffer` for the given text using the internal `FontSystem`.
102    /// This encapsulates the required `Metrics` and calls `set_text`.
103    pub fn create_buffer_for_text(&mut self, text: &str, text_style: TextStyle) -> Buffer {
104        let font_size = text_style.font_size;
105        let color = text_style.color;
106        let color = GlyphColor::rgba(color.0, color.1, color.2, color.3);
107        let align = text_style.text_align;
108        let weight = text_style.font_weight;
109        let style = text_style.font_style;
110        // reasonable default line height (1.2x)
111        let metrics = Metrics::relative(font_size, 1.2);
112
113        let mut buffer = Buffer::new(&mut self.font_sys, metrics);
114
115        // build attributes (defaults + color + metrics)
116        let attrs = Attrs::new()
117            .metrics(metrics)
118            .color(color)
119            .weight(Weight(weight.0))
120            .style(Style::from(style));
121
122        // shape and layout
123        buffer.set_text(
124            &mut self.font_sys,
125            text,
126            &attrs,
127            Shaping::Advanced,
128            Some(Align::from(align)),
129        );
130
131        buffer
132    }

134    /// 指定されたセクション群をギリフォン用の TextArea に変換して Atlas に転送する
135    pub fn queue<'a>(
136        &mut self,
137        device: &wgpu::Device,
138        queue: &wgpu::Queue,
139        sections: &'a [TextSection],
140    ) -> Result<(), PrepareError> {
141        // TextArea は Buffer を参照するライフタイムを持つため、一時的にベクタに詰めて渡す
142        let mut text_areas: Vec<TextArea<'a>> = Vec::with_capacity(sections.len());
143
144        for s in sections.iter() {
145            let bounds = TextBounds {
146                left: s.clip_origin.0.round() as i32,
147                top: s.clip_origin.1.round() as i32,
148                right: (s.clip_origin.0 + s.bounds.0).round() as i32,
149                bottom: (s.clip_origin.1 + s.bounds.1).round() as i32,
150            };
151
152            // デフォルト色は Buffer 内の属性が優先されるため適当で良い
153            let default_color = GlyphColor::rgba(0, 0, 0, 255);
154
155            let area = TextArea {
156                buffer: &s.buffer,
157                left: s.screen_position.0,
158                top: s.screen_position.1,
159                scale: 1.0,
160                bounds,
161                default_color,
162                custom_glyphs: &[],
163            };
164
165            text_areas.push(area);
166        }
167
168        self.brush.prepare(
169            device,
170            queue,
171            &mut self.font_sys,
172            &mut self.atlas,
173            &self.viewport,
174            text_areas,
175            &mut self.swash_cache,
176        )
177    }
178
179    /// ビューポート（解像度）を更新
180    pub fn resize_view(&mut self, width: f32, height: f32, queue: &wgpu::Queue) {
181        self.viewport.update(
182            queue,
183            Resolution {
184                width: width as u32,
185                height: height as u32,
186            },
187        );
188    }
189
190    /// フレームを描画
191    pub fn draw<'a>(&mut self, rpass: &mut wgpu::RenderPass<'a>) {
192        self.brush
193            .render(&self.atlas, &self.viewport, rpass)
194            .expect("PANIC: Text draw failed");
195    }

198impl From<FontStyle> for Style {
199    fn from(value: FontStyle) -> Self {
200        match value {
201            FontStyle::Normal => Style::Normal,
202            FontStyle::Italic => Style::Italic,
203            FontStyle::Oblique => Style::Oblique,
204        }
205    }
