69    /// 新しいGPUレンダラーを作成
70    pub async fn new(window: Arc<Window>, font_path: Option<&str>) -> Result<Self> {
71        let size = window.inner_size();
72        let scale_factor = window.scale_factor();
73
74        // GPUドライバとの通信インスタンス
75        // wgpuインスタンスの作成
76        let instance = wgpu::Instance::new(&wgpu::InstanceDescriptor {
77            backends: select_wgpu_backends(),
78            ..Default::default()
79        });
80
81        // OSウィンドウとGPUの描画対象（サーフェス）を関連付ける
82        // サーフェスの作成
83        let surface = instance.create_surface(window.clone())?;
84
85        // 利用可能なGPU（物理デバイス）アダプターの取得
86        let adapter = instance
87            .request_adapter(&wgpu::RequestAdapterOptions {
88                power_preference: wgpu::PowerPreference::default(),
89                compatible_surface: Some(&surface),
90                force_fallback_adapter: false,
91            })
92            .await?;
93
94        // デバイスとキューの作成
95        let (device, queue) = adapter
96            .request_device(&wgpu::DeviceDescriptor {
97                label: None,
98                required_features: wgpu::Features::empty(),
99                required_limits: wgpu::Limits::default(),
100                experimental_features: Default::default(),
101                memory_hints: wgpu::MemoryHints::default(),
102                trace: Default::default(),
103            })
104            .await?;
105
106        // サーフェス設定
107        // フレームバッファ設定（解像度・フォーマットなど）
108        let surface_caps = surface.get_capabilities(&adapter);
109        let surface_format = surface_caps
110            .formats
111            .iter()
112            .copied()
113            .find(|f| f.is_srgb())
114            .unwrap_or(surface_caps.formats[0]);
115
116        let config = wgpu::SurfaceConfiguration {
117            usage: wgpu::TextureUsages::RENDER_ATTACHMENT,
118            format: surface_format,
119            width: size.width,
120            height: size.height,
121            present_mode: surface_caps.present_modes[0],
122            alpha_mode: surface_caps.alpha_modes[0],
123            view_formats: vec![],
124            desired_maximum_frame_latency: 2,
125        };
126        surface.configure(&device, &config);
127
128        // シェーダーの読み込み
129        // シェーダーモジュールの作成
130        // vertex/fragment for main pipeline
131        let main_shader = device.create_shader_module(wgpu::ShaderModuleDescriptor {
132            label: Some("Main Shader"),
133            source: wgpu::ShaderSource::Wgsl(include_str!("shader/main.wgsl").into()),
134        });
135
136        // --- レンダーパイプライン（頂点→ピクセル変換のルール）の作成 ---
137        let render_pipeline_layout =
138            device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
139                label: Some("Render Pipeline Layout"),
140                bind_group_layouts: &[],
141                immediate_size: 0,
142            });
143
144        let render_pipeline = device.create_render_pipeline(&wgpu::RenderPipelineDescriptor {
145            label: Some("Render Pipeline"),
146            layout: Some(&render_pipeline_layout),
147            cache: None,
148            vertex: wgpu::VertexState {
149                module: &main_shader,
150                entry_point: Some("vs_main"),
151                buffers: &[Vertex::desc()],
152                compilation_options: wgpu::PipelineCompilationOptions::default(),
153            },
154            fragment: Some(wgpu::FragmentState {
155                module: &main_shader,
156                entry_point: Some("fs_main"),
157                targets: &[Some(wgpu::ColorTargetState {
158                    format: config.format,
159                    blend: Some(wgpu::BlendState::ALPHA_BLENDING),
160                    write_mask: wgpu::ColorWrites::ALL,
161                })],
162                compilation_options: wgpu::PipelineCompilationOptions::default(),
163            }),
164            primitive: wgpu::PrimitiveState {
165                topology: wgpu::PrimitiveTopology::TriangleList,
166                strip_index_format: None,
167                front_face: wgpu::FrontFace::Ccw,
168                cull_mode: None, // 三角扇がカリングで消えちゃう...
169                polygon_mode: wgpu::PolygonMode::Fill,
170                unclipped_depth: false,
171                conservative: false,
172            },
173            depth_stencil: None,
174            multisample: wgpu::MultisampleState {
175                count: 1,
176                mask: !0,
177                alpha_to_coverage_enabled: false,
178            },
179            multiview_mask: None,
180        });
181        // --- レンダーパイプライン作成終了 ---
182
183        // テキスト描画用ラッパーの初期化。引数で渡されたフォントパスがあればそれを優先して読み込む。
184        let text_renderer = if let Some(p) = font_path {
185            match std::fs::read(p) {
186                Ok(bytes) => {
187                    match TextRenderer::new_from_bytes(&device, &queue, config.format, bytes) {
188                        Ok(t) => Some(t),
189                        Err(e) => {
190                            log::warn!(target:"PRender::gpu::font" ,"failed to init text renderer from provided font: {}", e);
191                            None
192                        }
193                    }
194                }
195                Err(e) => {
196                    log::warn!(target:"PRender::gpu::font" ,"failed to read font path '{}': {}", p, e);
197                    None
198                }
199            }
200        } else {
201            match TextRenderer::new_from_device(&device, &queue, config.format) {
202                Ok(t) => Some(t),
203                Err(e) => {
204                    log::warn!(target:"PRender::gpu::font" ,"no system font found for text renderer: {}", e);
205                    None
206                }
207            }
208        };
209
210        // Enable text culling by default, allow override by env var
211        let enable_text_culling = std::env::var("ORINIUM_TEXT_CULL")
212            .map(|v| v != "0")
213            .unwrap_or(true);
214
215        Ok(Self {
216            surface,
217            device,
218            queue,
219            config,
220            size,
221            scale_factor,
222            render_pipeline,
223            vertex_buffer: None,
224            vertices: vec![],
225            num_vertices: 0,
226            text_renderer,
227            enable_text_culling,
228        })
229    }

231    /// ウィンドウサイズが変更された時の処理
232    pub fn resize(&mut self, new_size: winit::dpi::PhysicalSize<u32>) {
233        if new_size.width > 0 && new_size.height > 0 {
234            log::info!(target:"PRender::gpu::resized", "Resized: {}x{}", new_size.width, new_size.height);
235
236            let old_size = self.size;
237
238            self.size = new_size;
239
240            self.config.width = new_size.width;
241            self.config.height = new_size.height;
242
243            self.surface.configure(&self.device, &self.config);
244
245            self.update_vertices(old_size, new_size);
246
247            if let Some(tr) = &mut self.text_renderer {
248                tr.resize_view(
249                    self.config.width as f32,
250                    self.config.height as f32,
251                    &self.queue,
252                );
253            }
254        }
255    }

257    /// 描画命令を解析して頂点バッファやテキストキューに登録
258    pub fn parse_draw_commands(&mut self, commands: &[DrawCommand]) {
259        let screen_width = self.size.width as f32;
260        let screen_height = self.size.height as f32;
261
262        // --- 頂点データ ---
263        let mut vertices = Vec::new();
264        // --- Text ---
265        let mut sections: Vec<TextSection> = Vec::new();
266        // --- scale_factor ---
267        let sf = self.scale_factor as f32;
268        // --- transform stack ---
269        let mut transform_stack: Vec<(f32, f32)> = vec![(0.0, 0.0)];
270        let current_transform = |stack: &Vec<(f32, f32)>| -> (f32, f32) {
271            let mut dx = 0.0;
272            let mut dy = 0.0;
273            for (x, y) in stack.iter() {
274                dx += x;
275                dy += y;
276            }
277            (dx, dy)
278        };
279        // --- clip stack ---
280        #[derive(Clone, Copy)]
281        struct ClipRect {
282            x: f32,
283            y: f32,
284            w: f32,
285            h: f32,
286        }
287        let mut clip_stack: Vec<ClipRect> = vec![ClipRect {
288            x: 0.0,
289            y: 0.0,
290            w: screen_width,
291            h: screen_height,
292        }];
293        let current_clip = |stack: &Vec<ClipRect>| -> ClipRect { *stack.last().unwrap() };
294
295        for command in commands {
296            match command {
297                // Transform (Push / Pop)
298                DrawCommand::PushTransform { dx, dy } => {
299                    transform_stack.push((*dx, *dy));
300                }
301                DrawCommand::PopTransform => {
302                    if transform_stack.len() > 1 {
303                        transform_stack.pop();
304                    }
305                }
306
307                // Clip (Push / Pop)
308                DrawCommand::PushClip {
309                    x,
310                    y,
311                    width: w,
312                    height: h,
313                } => {
314                    let (tdx, tdy) = current_transform(&transform_stack);
315                    let new_clip = ClipRect {
316                        x: x + tdx,
317                        y: y + tdy,
318                        w: *w,
319                        h: *h,
320                    };
321
322                    // 現在の clip との AND を取る
323                    let parent = current_clip(&clip_stack);
324
325                    let x1 = new_clip.x.max(parent.x);
326                    let y1 = new_clip.y.max(parent.y);
327                    let x2 = (new_clip.x + new_clip.w).min(parent.x + parent.w);
328                    let y2 = (new_clip.y + new_clip.h).min(parent.y + parent.h);
329
330                    clip_stack.push(ClipRect {
331                        x: x1,
332                        y: y1,
333                        w: (x2 - x1).max(0.0),
334                        h: (y2 - y1).max(0.0),
335                    });
336                }
337
338                DrawCommand::PopClip => {
339                    if clip_stack.len() > 1 {
340                        clip_stack.pop();
341                    }
342                }
343
344                // Rectangle
345                DrawCommand::DrawRect {
346                    x,
347                    y,
348                    width: w,
349                    height: h,
350                    color,
351                } => {
352                    // transform
353                    let (tdx, tdy) = current_transform(&transform_stack);
354                    let mut x1 = (x + tdx) * sf;
355                    let mut y1 = (y + tdy) * sf;
356                    let mut x2 = x1 + w * sf;
357                    let mut y2 = y1 + h * sf;
358
359                    // clip 取得
360                    let clip = current_clip(&clip_stack);
361
362                    // 完全に外なら skip
363                    if x2 <= clip.x * sf
364                        || x1 >= (clip.x + clip.w) * sf
365                        || y2 <= clip.y * sf
366                        || y1 >= (clip.y + clip.h) * sf
367                    {
368                        continue;
369                    }
370
371                    // 部分クリップ
372                    x1 = x1.max(clip.x * sf);
373                    y1 = y1.max(clip.y * sf);
374                    x2 = x2.min((clip.x + clip.w) * sf);
375                    y2 = y2.min((clip.y + clip.h) * sf);
376
377                    // NDC
378                    let ndc = |v, max| (v / max) * 2.0 - 1.0;
379
380                    let px1 = ndc(x1, screen_width);
381                    let py1 = -ndc(y1, screen_height);
382                    let px2 = ndc(x2, screen_width);
383                    let py2 = -ndc(y2, screen_height);
384
385                    let color = color.to_linear_f32_array();
386
387                    #[rustfmt::skip]
388                    vertices.extend_from_slice(&[
389                        Vertex { position: [px1, py1, 0.0], color },
390                        Vertex { position: [px1, py2, 0.0], color },
391                        Vertex { position: [px2, py1, 0.0], color },
392
393                        Vertex { position: [px2, py1, 0.0], color },
394                        Vertex { position: [px1, py2, 0.0], color },
395                        Vertex { position: [px2, py2, 0.0], color },
396                    ]);
397                }
398
399                // Text
400                // TODO:
401                // - Clip 用の width （描画限界）と改行用の max_width を分けて扱う
402                DrawCommand::DrawText {
403                    x,
404                    y,
405                    text,
406                    style,
407                    max_width,
408                } => {
409                    let (tdx, tdy) = current_transform(&transform_stack);
410
411                    let clip = current_clip(&clip_stack);
412
413                    let tw = if (tdx + x + max_width) < (clip.x + clip.w) {
414                        (tdx + x + max_width) - clip.x
415                    } else {
416                        clip.w
417                    };
418
419                    let th = clip.h;
420
421                    let font_size = &style.font_size;
422
423                    // Text culling: if enabled and the text's bounding box is fully outside current clip, skip creating buffer
424                    let mut skip_text = false;
425                    if self.enable_text_culling {
426                        // compute screen-space bbox
427                        let sx1 = (x + tdx) * sf;
428                        let sy1 = (y + tdy) * sf;
429                        // if width/height are zero or NaN, estimate from font size and line count
430                        let est_w = if !tw.is_finite() || tw <= 0.0 {
431                            // fall back: estimate width as font_size * 10.0 * approximate_chars
432                            (*font_size * sf) * (text.len().max(1) as f32) * 0.5
433                        } else {
434                            tw * sf
435                        };
436                        let est_h = if !th.is_finite() || th <= 0.0 {
437                            // estimate height as font_size * 1.2 * lines
438                            (*font_size * sf) * 1.2 * (text.lines().count() as f32).max(1.0)
439                        } else {
440                            th * sf
441                        };
442                        let sx2 = sx1 + est_w;
443                        let sy2 = sy1 + est_h;
444
445                        let clip_l = clip.x * sf;
446                        let clip_t = clip.y * sf;
447                        let clip_r = (clip.x + clip.w) * sf;
448                        let clip_b = (clip.y + clip.h) * sf;
449
450                        if sx2 <= clip_l || sx1 >= clip_r || sy2 <= clip_t || sy1 >= clip_b {
451                            skip_text = true;
452                        }
453                    }
454
455                    if skip_text {
456                        continue;
457                    }
458
459                    // Use TextRenderer helper to create a Buffer with correct FontSystem handling
460                    let section = if let Some(tr) = &mut self.text_renderer {
461                        let mut render_text_style = *style;
462                        render_text_style.font_size = *font_size * sf;
463                        let buffer = tr.create_buffer_for_text(text, render_text_style);
464
465                        TextSection {
466                            screen_position: ((*x + tdx) * sf, (*y + tdy) * sf),
467                            clip_origin: (clip.x * sf, clip.y * sf),
468                            bounds: (tw * sf, th * sf),
469                            buffer,
470                        }
471                    } else {
472                        // No text renderer available; skip
473                        continue;
474                    };
475                    sections.push(section);
476                }
477
478                // Polygon
479                DrawCommand::DrawPolygon { points, color } => {
480                    // transform
481                    let (tdx, tdy) = current_transform(&transform_stack);
482                    let transformed_points: Vec<(f32, f32)> = points
483                        .iter()
484                        .map(|(px, py)| ((px + tdx) * sf, (py + tdy) * sf))
485                        .collect();
486
487                    // clip 取得
488                    let clip = current_clip(&clip_stack);
489                    // clip in scaled (screen) coords
490                    let clip_l = clip.x * sf;
491                    let clip_t = clip.y * sf;
492                    let clip_r = (clip.x + clip.w) * sf;
493                    let clip_b = (clip.y + clip.h) * sf;
494
495                    // Quick reject by bounding box
496                    let mut min_x = f32::INFINITY;
497                    let mut min_y = f32::INFINITY;
498                    let mut max_x = f32::NEG_INFINITY;
499                    let mut max_y = f32::NEG_INFINITY;
500                    for (x, y) in transformed_points.iter() {
501                        min_x = min_x.min(*x);
502                        min_y = min_y.min(*y);
503                        max_x = max_x.max(*x);
504                        max_y = max_y.max(*y);
505                    }
506                    if max_x <= clip_l || min_x >= clip_r || max_y <= clip_t || min_y >= clip_b {
507                        // fully outside
508                        continue;
509                    }
510
511                    // Helper: Sutherland–Hodgman polygon clipping against an axis-aligned edge
512                    let clip_against_edge = |poly: &Vec<(f32, f32)>, edge: u8| -> Vec<(f32, f32)> {
513                        // edge: 0=left,1=right,2=top,3=bottom
514                        let mut out: Vec<(f32, f32)> = Vec::new();
515                        if poly.is_empty() {
516                            return out;
517                        }
518                        let len = poly.len();
519                        for i in 0..len {
520                            let (sx, sy) = poly[i];
521                            let (ex, ey) = poly[(i + 1) % len];
522                            // inside test
523                            let inside = |x: f32, y: f32| -> bool {
524                                match edge {
525                                    0 => x >= clip_l, // left
526                                    1 => x <= clip_r, // right
527                                    2 => y >= clip_t, // top
528                                    3 => y <= clip_b, // bottom
529                                    _ => true,
530                                }
531                            };
532                            let s_in = inside(sx, sy);
533                            let e_in = inside(ex, ey);
534
535                            if s_in && e_in {
536                                // both inside
537                                out.push((ex, ey));
538                            } else if s_in && !e_in {
539                                // going out: add intersection
540                                // compute intersection between segment and clipping line
541                                let (ix, iy) = match edge {
542                                    0 | 1 => {
543                                        // vertical line x = clip_l or clip_r
544                                        let x_edge = if edge == 0 { clip_l } else { clip_r };
545                                        let dx = ex - sx;
546                                        if dx.abs() < f32::EPSILON {
547                                            (x_edge, sy)
548                                        } else {
549                                            let t = (x_edge - sx) / dx;
550                                            (x_edge, sy + t * (ey - sy))
551                                        }
552                                    }
553                                    2 | 3 => {
554                                        // horizontal line y = clip_t or clip_b
555                                        let y_edge = if edge == 2 { clip_t } else { clip_b };
556                                        let dy = ey - sy;
557                                        if dy.abs() < f32::EPSILON {
558                                            (sx, y_edge)
559                                        } else {
560                                            let t = (y_edge - sy) / dy;
561                                            (sx + t * (ex - sx), y_edge)
562                                        }
563                                    }
564                                    _ => (ex, ey),
565                                };
566                                out.push((ix, iy));
567                            } else if !s_in && e_in {
568                                // entering: add intersection then end point
569                                let (ix, iy) = match edge {
570                                    0 | 1 => {
571                                        let x_edge = if edge == 0 { clip_l } else { clip_r };
572                                        let dx = ex - sx;
573                                        if dx.abs() < f32::EPSILON {
574                                            (x_edge, sy)
575                                        } else {
576                                            let t = (x_edge - sx) / dx;
577                                            (x_edge, sy + t * (ey - sy))
578                                        }
579                                    }
580                                    2 | 3 => {
581                                        let y_edge = if edge == 2 { clip_t } else { clip_b };
582                                        let dy = ey - sy;
583                                        if dy.abs() < f32::EPSILON {
584                                            (sx, y_edge)
585                                        } else {
586                                            let t = (y_edge - sy) / dy;
587                                            (sx + t * (ex - sx), y_edge)
588                                        }
589                                    }
590                                    _ => (ex, ey),
591                                };
592                                out.push((ix, iy));
593                                out.push((ex, ey));
594                            } else {
595                                // both outside: do nothing
596                            }
597                        }
598                        out
599                    };
600
601                    // Triangulate polygon into fan triangles from vertex 0, clip each triangle, and push resulting triangles
602                    if transformed_points.len() < 3 {
603                        continue;
604                    }
605
606                    // NDC helper
607                    let ndc = |v: f32, max: f32| (v / max) * 2.0 - 1.0;
608
609                    let color_arr = color.to_linear_f32_array();
610
611                    let v0 = transformed_points[0];
612                    for i in 1..(transformed_points.len() - 1) {
613                        let tri = vec![v0, transformed_points[i], transformed_points[i + 1]];
614                        // clip triangle against rect using Sutherland–Hodgman (4 edges)
615                        let mut poly = tri;
616                        poly = clip_against_edge(&poly, 0); // left
617                        if poly.is_empty() {
618                            continue;
619                        }
620                        poly = clip_against_edge(&poly, 1); // right
621                        if poly.is_empty() {
622                            continue;
623                        }
624                        poly = clip_against_edge(&poly, 2); // top
625                        if poly.is_empty() {
626                            continue;
627                        }
628                        poly = clip_against_edge(&poly, 3); // bottom
629                        if poly.is_empty() {
630                            continue;
631                        }
632
633                        // triangulate resulting polygon as fan
634                        for j in 1..(poly.len() - 1) {
635                            let p1 = poly[0];
636                            let p2 = poly[j];
637                            let p3 = poly[j + 1];
638
639                            let px1 = ndc(p1.0, screen_width);
640                            let py1 = -ndc(p1.1, screen_height);
641                            let px2 = ndc(p2.0, screen_width);
642                            let py2 = -ndc(p2.1, screen_height);
643                            let px3 = ndc(p3.0, screen_width);
644                            let py3 = -ndc(p3.1, screen_height);
645
646                            vertices.push(Vertex {
647                                position: [px1, py1, 0.0],
648                                color: color_arr,
649                            });
650                            vertices.push(Vertex {
651                                position: [px2, py2, 0.0],
652                                color: color_arr,
653                            });
654                            vertices.push(Vertex {
655                                position: [px3, py3, 0.0],
656                                color: color_arr,
657                            });
658                        }
659                    }
660                }
661
662                // Ellipse
663                #[allow(unused)]
664                DrawCommand::DrawEllipse {
665                    center,
666                    radius_x,
667                    radius_y,
668                    color,
669                } => {
670                    // transform
671                    let (tdx, tdy) = current_transform(&transform_stack);
672                    let cx = center.0 + tdx;
673                    let cy = center.1 + tdy;
674
675                    // clip 取得
676                    let clip = current_clip(&clip_stack);
677
678                    todo!("Ellipse drawing with clipping is not implemented yet");
679                }
680            }
681        }
682
683        self.set_vertex_buffer(vertices);
684
685        // テキストセクションをキューに追加
686        if let Some(tr) = &mut self.text_renderer {
687            tr.queue(&self.device, &self.queue, &sections).unwrap();
688        }
689    }

691    /// フレームを描画
692    pub fn render(&mut self) -> Result<()> {
693        // 描画するフレームバッファを取得
694        let output = self.surface.get_current_texture()?;
695        let view = output
696            .texture
697            .create_view(&wgpu::TextureViewDescriptor::default());
698
699        // アニメーション中はテキストブラシが更新位置を反映できるようにセクションを再キューする必要がある
700        // 補足: 呼び出し元（UI層）も各フレームで描画コマンドを再キューしているため、ここではアニメーション状態を返り値で通知するだけ
701
702        // GPUコマンドのエンコーダーの作成
703        let mut encoder = self
704            .device
705            .create_command_encoder(&wgpu::CommandEncoderDescriptor {
706                label: Some("Render Encoder"),
707            });
708
709        // 描画パスの開始
710        {
711            let mut render_pass = encoder.begin_render_pass(&wgpu::RenderPassDescriptor {
712                label: Some("Render Pass"),
713                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
714                    view: &view,
715                    resolve_target: None,
716                    ops: wgpu::Operations {
717                        // 背景色をクリア
718                        load: wgpu::LoadOp::Clear(wgpu::Color {
719                            r: 1.0,
720                            g: 1.0,
721                            b: 1.0,
722                            a: 1.0,
723                        }),
724                        store: wgpu::StoreOp::Store,
725                    },
726                    depth_slice: None,
727                })],
728                depth_stencil_attachment: None,
729                occlusion_query_set: None,
730                timestamp_writes: None,
731                multiview_mask: None,
732            });
733
734            // 使用するシェーダー・設定をセット
735            render_pass.set_pipeline(&self.render_pipeline);
736            // 頂点バッファをセットして描画
737            if let Some(ref vertex_buffer) = self.vertex_buffer {
738                render_pass.set_vertex_buffer(0, vertex_buffer.slice(..));
739                render_pass.draw(0..self.num_vertices, 0..1);
740            }
741        }
742
743        // テキストをレンダリング
744        if let Some(tr) = &mut self.text_renderer {
745            let mut rpass = encoder.begin_render_pass(&wgpu::RenderPassDescriptor {
746                label: Some("Text Render Pass"),
747                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
748                    view: &view,
749                    resolve_target: None,
750                    ops: wgpu::Operations {
751                        load: wgpu::LoadOp::Load,
752                        store: wgpu::StoreOp::Store,
753                    },
754                    depth_slice: None,
755                })],
756                depth_stencil_attachment: None,
757                occlusion_query_set: None,
758                timestamp_writes: None,
759                multiview_mask: None,
760            });
761            tr.draw(&mut rpass);
762        }
763
764        // コマンドをGPUに送信
765        self.queue.submit(std::iter::once(encoder.finish()));
766
767        // フレームを画面に表示
768        output.present();
769
770        Ok(())
771    }

773    fn update_vertices(
774        &mut self,
775        old_size: winit::dpi::PhysicalSize<u32>,
776        new_size: winit::dpi::PhysicalSize<u32>,
777    ) {
778        let old_w = old_size.width as f32;
779        let old_h = old_size.height as f32;
780        let new_w = new_size.width as f32;
781        let new_h = new_size.height as f32;
782
783        let mut new_vertices = self.vertices.clone();
784
785        for vertex in new_vertices.iter_mut() {
786            // old NDC -> logical
787            let logical_x = (vertex.position[0] + 1.0) / 2.0 * old_w;
788            let logical_y = -(vertex.position[1] - 1.0) / 2.0 * old_h;
789
790            // logical -> new NDC
791            vertex.position[0] = (logical_x / new_w) * 2.0 - 1.0;
792            vertex.position[1] = -((logical_y / new_h) * 2.0 - 1.0);
793        }
794        self.set_vertex_buffer(new_vertices);
795    }
796
797    fn set_vertex_buffer(&mut self, vertices: Vec<Vertex>) {
798        // 頂点バッファを登録
799        if !vertices.is_empty() {
800            self.vertex_buffer = Some(self.device.create_buffer_init(
801                &wgpu::util::BufferInitDescriptor {
802                    label: Some("Vertex Buffer"),
803                    contents: bytemuck::cast_slice(&vertices),
804                    usage: wgpu::BufferUsages::VERTEX,
805                },
806            ));
807            self.num_vertices = vertices.len() as u32;
808        }
809        self.vertices = vertices;
810    }

817fn select_wgpu_backends() -> wgpu::Backends {
818    if let Ok(value) = env::var("ORINIUM_WGPU_BACKEND") {
819        match value.to_lowercase().as_str() {
820            "gl" | "opengl" => return wgpu::Backends::GL,
821            "vulkan" | "vk" => return wgpu::Backends::VULKAN,
822            "metal" => return wgpu::Backends::METAL,
823            "dx12" | "d3d12" => return wgpu::Backends::DX12,
824            "primary" => return wgpu::Backends::PRIMARY,
825            _ => {}
826        }
827    }
828
829    let is_wsl = env::var_os("WSL_DISTRO_NAME").is_some() || env::var_os("WSL_INTEROP").is_some();
830    let is_wayland = env::var_os("WAYLAND_DISPLAY").is_some();
831
832    if is_wsl && is_wayland {
833        // WSLg + Wayland can be unstable with Vulkan; prefer GL by default.
834        return wgpu::Backends::GL;
835    }
836
837    wgpu::Backends::PRIMARY
838}
